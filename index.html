## 2. install gemini-cli
1. update pkg
 $ sudo apt update
 $ sudo apt upgrade -y
2. nvm install 
  $ curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.39.5/install.sh | bash
  $ exit
* install node
 $ nvm install node 
 $ nvm use node 
 $ nvm alias default node # set default node version to latest for new shells
 $ node --version

* install gemini-cli 
  $ npm install -g @google/gemini-cli
  $ gemini --version 
* start gemini
 $ gemini
